this

Calling context

Depends not upon the function, but upon how you call the function. Think of it as an arugment that is implicitly passed to the function by the runtime.

Global calling context example


function bar() {
  console.log(this);
}

bar();

Member calling context example


var foo = {};

function bar() {
    console.log(this === foo);
}

foo.bar = bar;
foo.bar();

Why calling context? (1)


var foo = {
    bar: 123,
    bas: function () {
        console.log('inside this.bar is:', this.bar);
    }
}

foo.bas();

Why calling context? (2)


function bar() {
    console.log('inside this.foo is:', this.foo);
}

var a = { foo: 123, bar: bar };
var b = { foo: 456, bar: bar };

a.bar();
b.bar();

Why calling context? (3)


function bar() {
    console.log('inside this.foo is:', this.foo);
}

var a = { foo: 123 };
a.bar = bar;

a.bar();

Same functions?


var foo = {};
var bar = {};
log(foo.toString == bar.toString);

Are these the same functions? Where are they coming from? (Object.__proto__.toString)

Same proto


var foo = {};
var bar = {};
log(foo.__proto__ == bar.__proto__);

Share the proto

Same toString


var foo = {};
var bar = {};
log(foo.__proto__.toString == bar.__proto__.toString);

__proto__ lookup


var foo = {};
print(foo.__proto__);
print(foo.__proto__.__proto__);

print('----');
foo.__proto__.bar = 123;
print(foo.bar);

print('----');
foo.bar = 456;
print(foo.bar);

print('----');
delete foo.bar;
print(foo.bar);

Since proto is shared don't do this!

__proto__ chain


var foo = {
    __proto__: {
        __proto__: {
            __proto__: {
                bar: 123
            }
        }
    }
};

print(foo.bar);

Extend the language


var objectProto = ({}).__proto__;
objectProto.log = function() {
    print(this);
};

var foo = {
    a: 123
};
var bar = {
    b: 456
};

foo.log();
bar.log();

Now you shouldn't do this deep in your library or you are bound to break other libraries and possibly future JavaScript

The effects of `new`

The prototype property


function Foo() { }
print(Foo.prototype);

// Has a default member
print(Foo.prototype.constructor === Foo);

Non enumerated constructor property

Creates a new object


function Foo() {
}

// without the new operator
print(Foo());

// with the new operator
print(new Foo());

Effect on this


function Foo() {
    this.bar = 123;
    print('Is this global?: '+ (this == window));
}

// without the new operator
Foo();
print(window.bar);

// with the new operator
var newFoo = new Foo();
print(newFoo.bar);

prototype and __proto__


function Foo() { }

var foo = new Foo();

print(foo.__proto__ === Foo.prototype);

prototype advantage


function Foo() {}
Foo.prototype.log = function() {
    print('log');
}

var a = new Foo();
var b = new Foo();

a.log();
b.log();

Provides a standard way of creating shared functions

Basic Class Structure


function Foo(val) {
    this.val = val;
}
Foo.prototype.log = function () {
    print(this.val);
}

var a = new Foo(1);
var b = new Foo(2);

a.log();
b.log();

Each gets its own `this` and proto (pointing to prototype) is shared we get an efficient class structure. The `this` inside the prototype members is depended upon the calling context

Global State

Set data-state="something" on a slide and "something" will be added as a class to the document element when the slide is open. This lets you apply broader style changes, like switching the background.

Custom Events

Additionally custom events can be triggered on a per slide basis by binding to the data-state name. 


Reveal.addEventListener( 'customevent', function() {
  console.log( '"customevent" has fired' );
} );
